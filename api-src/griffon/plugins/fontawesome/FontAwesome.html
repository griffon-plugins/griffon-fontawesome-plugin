623     public static FontAwesome findByDescription(@Nonnull String description) {
624         requireNonBlank(description, "Icon description must not be blank.");
625         String[] parts = description.split(":");
626         for (FontAwesome font : values()) {

628                 return font;

630         }
631         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
632     }
